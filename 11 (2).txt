    <script>

        // ===================================================================

        const publicView = document.getElementById('public-view');

        // ===================================================================
        //                            UTILITY FUNCTIONS
        // ===================================================================
        const showLoading = () => {
            loadingSpinner.style.display = 'flex';
            document.body.style.overflow = 'hidden';
        };
        
        const hideLoading = () => {
            loadingSpinner.style.display = 'none';
            document.body.style.overflow = 'auto';
        };

        const showModal = (id) => {
            document.getElementById(id).style.display = 'flex';
            document.body.style.overflow = 'hidden';
        };
        
        const closeModal = (id) => {
            document.getElementById(id).style.display = 'none';
            document.body.style.overflow = 'auto';
        };

        // ===================================================================
        //                            VIEW MANAGEMENT
        // ===================================================================
        function showPublicView() {
            publicView.classList.remove('hidden');
            adminView.classList.add('hidden');
            imageView.classList.add('hidden');
            adminLoginBtn.classList.remove('hidden');
            logoutBtn.classList.add('hidden');
            isAdmin = false;
            loadFolders();
        }
        
        function showAdminInterface() {
            publicView.classList.add('hidden');
            adminView.classList.remove('hidden');
            imageView.classList.add('hidden');
            adminLoginBtn.classList.add('hidden');
            logoutBtn.classList.remove('hidden');
            isAdmin = true;
            loadFoldersForAdmin();
        }
        
        function showImageView(folderId, folderName, folderDriveId) {
            currentAlbum = { id: folderId, name: folderName, driveId: folderDriveId };
            publicView.classList.add('hidden');
            adminView.classList.add('hidden');
            imageView.classList.remove('hidden');
            document.getElementById('album-title').textContent = folderName;
            loadImages(folderId);
        }

        // ===================================================================
        //                            DATA FETCHING & RENDERING
        // ===================================================================

        // FOLDERS
        function loadFolders() {
            showLoading();
            const sortBy = document.getElementById('sort-folders').value;
            google.script.run.withSuccessHandler(folders => {
                currentFolders = folders;
                renderFolders(folders);
                if (isAdmin) {
                    renderAdminFolderList(folders);
                    renderFolderSelectForUpload(folders);
                }
                hideLoading();
            }).withFailureHandler(err => {
                console.error(err);
                Swal.fire({
                    title: 'Error',
                    text: 'ไม่สามารถโหลดข้อมูลอัลบั้มได้',
                    icon: 'error',
                    confirmButtonColor: '#3b82f6',
                });
                hideLoading();
            }).getFolders(sortBy);
        }

        function renderFolders(folders) {
            const grid = document.getElementById('folder-grid');
            grid.innerHTML = '';
            
            if (folders.length === 0) {
                grid.innerHTML = `

        // Helper function to format date
        function formatDate(dateString) {
            if (!dateString) return '';
            const options = { year: 'numeric', month: 'short', day: 'numeric' };
            return new Date(dateString).toLocaleDateString('th-TH', options);
        }

        // --- Image Upload Logic ---
        function setupUploadListeners() {
            const dropZone = document.getElementById('drop-zone');
            const fileInput = document.getElementById('file-input');

            dropZone.onclick = () => fileInput.click();

            dropZone.addEventListener('dragover', e => {
                e.preventDefault();
                dropZone.classList.add('drop-zone--over');
            });

            ['dragleave', 'dragend'].forEach(type => {
                dropZone.addEventListener(type, e => {
                    dropZone.classList.remove('drop-zone--over');
                });
            });

            dropZone.addEventListener('drop', e => {
                e.preventDefault();
                if (e.dataTransfer.files.length) {
                    fileInput.files = e.dataTransfer.files;
                    handleFiles(fileInput.files);
                }
                dropZone.classList.remove('drop-zone--over');
            });

            fileInput.addEventListener('change', e => {
                if(fileInput.files.length) {
                    handleFiles(fileInput.files);
                }
            });
        }

        function handleFiles(files) {
            if (!currentAlbum.id || !currentAlbum.driveId) {
                Swal.fire({
                    title: 'ผิดพลาด',
                    text: 'กรุณาเลือกอัลบั้มก่อนอัปโหลด',
                    icon: 'warning',
                    confirmButtonColor: '#3b82f6',
                });
                return;
            }
            
            showLoading();
            const filesArray = Array.from(files);
            let uploadedCount = 0;
            let errorCount = 0;

            // Show upload progress
            const progressToast = Swal.mixin({
                toast: true,
                position: 'top-end',
                showConfirmButton: false,
                timerProgressBar: true,
                didOpen: (toast) => {
                    toast.addEventListener('mouseenter', Swal.stopTimer);
                    toast.addEventListener('mouseleave', Swal.resumeTimer);
                }
            });
            
            progressToast.fire({
                title: 'กำลังอัปโหลด...',
                html: `0/${filesArray.length} ไฟล์`,
                icon: 'info',
                timer: 20000,
            });

            filesArray.forEach(file => {
                const reader = new FileReader();
                reader.onload = (e) => {
                    const fileInfo = {
                        base64: e.target.result.split(',')[1],
                        mimeType: file.type,
                        name: file.name
                    };
                    google.script.run.withSuccessHandler(res => {
                        uploadedCount++;
                        // Update progress
                        Swal.update({
                            html: `${uploadedCount}/${filesArray.length} ไฟล์<br>
                                  <small>${errorCount} ไฟล์มีปัญหา</small>`
                        });
                        
                        if(uploadedCount + errorCount === filesArray.length) {
                            Swal.close();
                            if (errorCount > 0) {
                                Swal.fire({
                                    title: 'อัปโหลดเสร็จสิ้น',
                                    html: `อัปโหลดสำเร็จ ${uploadedCount} ไฟล์<br>
                                           มีปัญหา ${errorCount} ไฟล์`,
                                    icon: uploadedCount > 0 ? 'warning' : 'error',
                                    confirmButtonColor: '#3b82f6',
                                });
                            } else {
                                Swal.fire({
                                    title: 'สำเร็จ!',
                                    text: `อัปโหลด ${uploadedCount} รูปภาพเรียบร้อย`,
                                    icon: 'success',
                                    confirmButtonColor: '#3b82f6',
                                });
                            }
                            loadImagesForAdmin(currentAlbum.id); // Refresh list
                            hideLoading();
                        }
                    }).withFailureHandler(err => {
                        console.error(err);
                        errorCount++;
                        // Update progress
                        Swal.update({
                            html: `${uploadedCount}/${filesArray.length} ไฟล์<br>
                                  <small>${errorCount} ไฟล์มีปัญหา</small>`
                        });
                        
                        if(uploadedCount + errorCount === filesArray.length) {
                            Swal.close();
                            if (errorCount > 0) {
                                Swal.fire({
                                    title: 'อัปโหลดเสร็จสิ้น',
                                    html: `อัปโหลดสำเร็จ ${uploadedCount} ไฟล์<br>
                                           มีปัญหา ${errorCount} ไฟล์`,
                                    icon: uploadedCount > 0 ? 'warning' : 'error',
                                    confirmButtonColor: '#3b82f6',
                                });
                            }
                            loadImagesForAdmin(currentAlbum.id); // Refresh list
                            hideLoading();
                        }
                    }).uploadImage(fileInfo, currentAlbum.driveId, currentAlbum.id);
                };
                reader.onerror = () => {
                    errorCount++;
                    // Update progress
                    Swal.update({
                        html: `${uploadedCount}/${filesArray.length} ไฟล์<br>
                               <small>${errorCount} ไฟล์มีปัญหา</small>`
                    });
                    
                    if(uploadedCount + errorCount === filesArray.length) {
                        Swal.close();
                        if (errorCount > 0) {
                            Swal.fire({
                                title: 'อัปโหลดเสร็จสิ้น',
                                html: `อัปโหลดสำเร็จ ${uploadedCount} ไฟล์<br>
                                       มีปัญหา ${errorCount} ไฟล์`,
                                icon: uploadedCount > 0 ? 'warning' : 'error',
                                confirmButtonColor: '#3b82f6',
                            });
                        }
                        loadImagesForAdmin(currentAlbum.id); // Refresh list
                        hideLoading();
                    }
                };
                reader.readAsDataURL(file);
            });
        }
